# Basic

- order: 0

read the option
:::
---

````jsx
import { Select } from '@alifd/next';

const Option = Select.Option;

class Demo extends React.Component {
    state = {
        reader: ''
    }
    onToggleHighlightItem = (item, type) => {
        this.setState({
            reader: item.label
        });
    };
    render() {
        return <div>
            <span className="next-visually-hidden" aria-live="assertive">{this.state.reader}</span>
             <Select id="basic-demo" onToggleHighlightItem={this.onToggleHighlightItem} defaultValue="jack" showSearch hasClear>
                <Option value="jack">Jack</Option>
                <Option value="frank">Frank</Option>
                <Option value="hugo">Hugo</Option>
            </Select>
        </div>
    }
}

ReactDOM.render(<Demo/>, mountNode);
````
